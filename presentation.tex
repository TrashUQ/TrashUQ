            \textbf{MQTT Telemetry}
            \begin{itemize}
                \item Real-time status, metrics, and classification streams.
                \item Low-bandwidth, high-reliability pub/sub.
            \end{itemize}
            \vspace{0.3cm}
            \textbf{gRPC Coordination}
            \begin{itemize}
                \item High-performance binary RPC for model weight transfer.
                \item Concurrency-safe: Rejects stale updates during rounds.
            \end{itemize}

\begin{frame}{Part B: Scaling Performance}
    \begin{itemize}

        \item \textbf{Condition}: Non-IID Dirichlet partitioning ($\alpha=0.3$).
        \item \textbf{Convergence}: Stable convergence with \textbf{93-94\% final accuracy}.
        \item \textbf{Efficiency}: predictable linear growth in communication cost.
